def save
  if persisted?
    requires :id
    data = service.update_volume(id, attributes).body
  else
    requires :name, :computing_cell, :billing_group, :size_in_gb, :iops, :iops_max
    data = service.create_volume(attributes).body
  end
  merge_attributes(data)
  true
end
